<SampleDescription>
    <p>
        The example provided below illustrates the process of using custom editors in place of the default editors in a Blazor Data Form component.
    </p>
</SampleDescription>

<ActionDescription>
    <p>
        The Blazor Data Form component provides default editors component based on the type of the field. You can also override this through the <code>EditorType</code> property. The Data Form component is equipped with the following editor options.
    </p>
    <ul>
        <li><code>TextBox</code></li>
        <li><code>TextArea</code></li>
        <li><code>DropdownList</code></li>
        <li><code>ComboBox</code></li>
        <li><code>AutoComplete</code></li>
        <li><code>Checkbox</code></li>
        <li><code>DatePicker</code></li>
        <li><code>DateTimePicker</code></li>
        <li><code>TimePicker</code></li>
        <li><code>Password</code></li>
        <li><code>Switch</code></li>
    </ul>
</ActionDescription>

<div class="col-lg-12 control-section">
   
    <div class="control-wrapper">
        <h4 class="control-header">Hotel Booking</h4>
        <SfDataForm ID="MyForm"
                    Model="@HotelBookingModel"
                    ColumnCount="2"
                    ColumnSpacing="20px"
                    ButtonsAlignment="FormButtonsAlignment.Right">
            <FormValidator>
                <DataAnnotationsValidator></DataAnnotationsValidator>
            </FormValidator>
            <FormItems>
                <FormItem Field="@nameof(HotelBookingModel.FirstName)" ColumnSpan="1" Placeholder="Enter first name"/>
                <FormItem Field="@nameof(HotelBookingModel.LastName)" ColumnSpan="1" Placeholder="Enter last name"/>
                <FormItem Field="@nameof(HotelBookingModel.PhoneNumber)" ColumnSpan="2">
                    <Template>
                        <label class="e-form-label">Phone Number</label>
                        <SfMaskedTextBox Mask="+(00) 0000000000" Placeholder="Enter phone number" @bind-Value="HotelBookingModel.PhoneNumber"></SfMaskedTextBox>
                    </Template>
                </FormItem>
                <FormItem Field="@nameof(HotelBookingModel.Email)" ColumnSpan="2" Placeholder="someone@example.com" />
                <FormItem Field="@nameof(HotelBookingModel.CheckIn)" EditorType="FormEditorType.DateTimePicker" ColumnSpan="1">
                </FormItem>
                <FormItem Field="@nameof(HotelBookingModel.CheckOut)" ColumnSpan="1">
                    <Template>
                        <label class="e-form-label">Check Out</label>
                        <SfDateTimePicker TValue="DateTime?" @bind-Value="HotelBookingModel.CheckOut" Placeholder="Enter check out date" Min="HotelBookingModel.CheckIn "></SfDateTimePicker>
                    </Template>
                </FormItem>
                <FormItem Field="@nameof(HotelBookingModel.AdultCount)" ColumnSpan="1" Placeholder="Enter adults count"/>
                <FormItem Field="@nameof(HotelBookingModel.RoomPreferences)" ColumnSpan="1" EditorType="FormEditorType.ComboBox"/>
                <FormItem Field="@nameof(HotelBookingModel.Address)" ColumnSpan="2" EditorType="FormEditorType.TextArea"/>
                <FormItem Field="@nameof(HotelBookingModel.Accept)" ColumnSpan="2" EditorType="FormEditorType.Switch"/>
                </FormItems>
        </SfDataForm>
    </div>
</div>
